The source data is read into the workspace:

PUCC_data <- read.csv("vt_pucc_issue_DL_23_03_2022.csv")

A subset of the data with only the columns needed to measure the age and the emission parameters

PUCC_plot_data = subset(PUCC_data, select = c(9,13:27,41:44) )

As the registration date is given as a character, we can change it to the actual date and extract the year from it.

PUCC_plot_data$regn_dt <- as.Date(PUCC_plot_data$regn_dt)
PUCC_plot_data$regn_yr <- as.numeric(format(PUCC_plot_data$regn_dt, "%Y"))

The age distribution of the vehicles in total

hist(PUCC_plot_data$regn_yr)

Age_all <- PUCC_plot_data %>% count(regn_yr,sort=TRUE)

To filter out the different fuel types

PUCC_petrol <- filter(PUCC_plot_data, (prescribed_high_idle_emission_limit_co>0 | prescribed_idle_emission_limit_hc>0) & lamda_rule_applicable=="t")

PUCC_CNG <- filter(PUCC_plot_data, (prescribed_high_idle_emission_limit_co>0 | prescribed_idle_emission_limit_hc>0) & lamda_rule_applicable=="f")

PUCC_diesel <-filter(PUCC_plot_data, prescribed_light_absorption_cofeeicient>0)
